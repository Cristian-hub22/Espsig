Función sumatoria <- sumapares(N)
	Definir i, num, inicio Como Entero
	sumatoria <- 0
	Si N % 2=0 Entonces
		inicio <- N+2
	SiNo
		inicio <- N+1
	FinSi
	Para i=1 Hasta N Con Paso 1 Hacer
		num <- inicio+(i-1)*2
		sumatoria <- sumatoria+num
	FinPara
FinFunción
